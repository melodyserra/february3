<script>

  number = Math.ceil(Math.random()*5)

  guess = prompt("I'm thinking of a number between 1 and 5. Can you guess what number it is?")

  while (guess != number){
    guess = prompt("Sorry, guess again!")
  }

  alert("That's Correct!")

</script>
